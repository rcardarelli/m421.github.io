Partitioning For Model

hospital4 <- hospital3 %>% select(-Season, -Method_of_Pmt, -LOS) %>% drop_na()
hospital4 <- hospital4[sample(1:nrow(hospital4), 20000),]

#imbalanced
prop.table(table(hospital4$target))

library(caret)
## Loading required package: lattice
## 
## Attaching package: 'caret'
## The following object is masked from 'package:purrr':
## 
##     lift
#partitioning the data
index = createDataPartition(y=hospital4$target, p=0.8, list=FALSE)

train = hospital4[index,]
test = hospital4[-index,]

#undersampling on training
train0 = train[train$target == "0",]
train1 = train[train$target == "1",]
n0 = nrow(train0)
n1 = nrow(train1)
train00 = train0[sample(1:n0, n1),]
train_under = rbind(train00, train1)

#checked to make sure more points in undersampled training than in testing
length(test$target) < length(train_under$target)
## [1] TRUE

ranger

-mtry: Number of variables to possibly split at in each node. Default is the (rounded down) square root of the number variables. -min.node.size: Minimal node size. -splitrule: Split rule. -best with min.node.size = 2

myGrid = expand.grid(mtry = c(1:3), splitrule = c("gini"), min.node.size = c(2,3))
model_under = train(target~., data = train_under, method = "ranger", tuneGrid = myGrid)
pred = predict(model_under, test)
accuracy = confusionMatrix(table(pred, test$target))$overall[[1]]
ba = confusionMatrix(table(pred, test$target))$byClass[[11]]
plot(model_under)

rpart

  • CP: complexity parameter
  • documentation “The main role of this parameter is to save computing time by pruning off splits that are obviously not worthwhile. Essentially,the user informs the program that any split which does not improve the fit by cp will likely be pruned off by cross-validation, and that hence the program need not pursue it.”
myGrid = expand.grid(cp=seq(.01,.05,by=.01))
model_under = train(target~., data = train_under, method = "rpart",tuneGrid = myGrid)
pred  = predict(model_under, test)
